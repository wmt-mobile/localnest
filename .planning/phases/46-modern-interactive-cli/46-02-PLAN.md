**2. Replace `printText(results)` function entirely:**

Before:
```js
function printText(results) {
  console.log('LocalNest Doctor');
  console.log('');

  for (const r of results) {
    const mark = r.ok ? 'OK' : 'FAIL';
    console.log(`[${mark}] ${r.id}: ${r.detail}`);
    if (!r.ok && r.fix) {
      console.log(`  fix: ${r.fix}`);
    }
  }

  const failed = results.filter((r) => !r.ok).length;
  console.log('');
  if (failed === 0) {
    console.log('Doctor result: healthy');
  } else {
    console.log(`Doctor result: ${failed} issue(s) found`);
  }
}
```

After:
```js
function printText(results) {
  console.log(c.bold('LocalNest Doctor'));
  console.log('');

  for (const r of results) {
    const mark = r.ok ? symbol.ok() : symbol.fail();
    console.log(`${mark} ${c.bold(r.id)}: ${r.detail}`);
    if (!r.ok && r.fix) {
      console.log(`   ${c.yellow('fix:')} ${r.fix}`);
    }
  }

  const passed = results.filter((r) => r.ok).length;
  const failed = results.length - passed;
  console.log('');
  console.log(`Health: ${bar(passed, results.length)}`);
  console.log('');
  if (failed === 0) {
    console.log(`${symbol.ok()} ${c.green('Doctor result: healthy')}`);
  } else {
    console.log(`${symbol.fail()} ${c.red(`Doctor result: ${failed} issue(s) found`)}`);
  }
}
```

**3. The `if (asJson)` branch in `main()` MUST remain completely unchanged.** Do NOT add any ANSI codes or imports into that branch. The JSON output must be machine-readable.

**4. `printHelp()` — optionally style the header only:**

function printSuccess(packageRef, indexConfig) {

  console.log(box.row(c.bold('LocalNest Setup Complete ✓'), 60));

  console.log(box.row(`Config:   ${configPath}`, 60));

    console.log(box.divider(60));
    console.log(box.row('Auto-installed into:', 60));
    for (const result of clientInstall.installed) {
      console.log(box.row(`  ${symbol.ok()} ${result.tool}: ${result.status}`, 60));
    }
  }
  console.log(box.bottom(60));
  console.log('');

  if (clientInstall.unsupported.length > 0) {
    console.log(c.yellow('Detected but not auto-configured:'));
    for (const item of clientInstall.unsupported) {

  console.log('2) If your tool was not updated automatically, copy this config:');

  console.log(JSON.stringify(preferredSnippet, null, 2));
  console.log('');
  console.log(rule());
