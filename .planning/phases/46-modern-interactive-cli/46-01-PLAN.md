// src/cli/ansi.ts
// Shared ANSI terminal utility for LocalNest CLI scripts.
// All color/symbol/box output is guarded by NO_COLOR env var and TTY detection.
// NO new runtime deps — uses built-in ANSI escape codes only.

/**
 * Returns true only when output is an interactive TTY AND NO_COLOR is not set.
 * Per https://no-color.org: any presence of NO_COLOR (even empty string) disables color.
 */
export function isColorEnabled(): boolean {
  return process.stdout.isTTY === true && process.env.NO_COLOR === undefined;
}

const esc = (code: number, s: string): string =>
  isColorEnabled() ? `\x1b[${code}m${s}\x1b[0m` : s;

/** Color functions — return plain text when color is disabled. */
export const c = {
  red:    (s: string) => esc(31, s),
  green:  (s: string) => esc(32, s),
  yellow: (s: string) => esc(33, s),
  cyan:   (s: string) => esc(36, s),
  bold:   (s: string) => esc(1,  s),
  dim:    (s: string) => esc(2,  s),
  gray:   (s: string) => esc(90, s),
};

/** Status symbols — colored unicode in TTY, plain ASCII fallback. */
export const symbol = {
  ok:   () => isColorEnabled() ? c.green('✓') : 'OK',
  fail: () => isColorEnabled() ? c.red('✗')   : 'FAIL',
  warn: () => isColorEnabled() ? c.yellow('⚠') : 'WARN',
  info: () => isColorEnabled() ? c.cyan('ℹ')   : 'INFO',
};

const isTTY = () => process.stdout.isTTY === true;

/** Box drawing for summary panels. Width = inner content width (default 62). */
export const box = {
  top: (w = 62): string =>
    isTTY() ? `╔${'═'.repeat(w)}╗` : `+${'-'.repeat(w)}+`,

  bottom: (w = 62): string =>
    isTTY() ? `╚${'═'.repeat(w)}╝` : `+${'-'.repeat(w)}+`,

  row: (text = '', w = 62): string => {
    const inner = w - 2; // 2 spaces padding
    const pad = Math.max(0, inner - text.length);
    return isTTY()
      ? `║  ${text}${' '.repeat(pad)}║`
      : `|  ${text}${' '.repeat(pad)}|`;
  },

  divider: (w = 62): string =>
    isTTY() ? `╟${'─'.repeat(w)}╢` : `|${'-'.repeat(w)}|`,
};

/**
 * Progress/health bar.
 * TTY:     "████░░░░░░ 3/10"
 * non-TTY: "[###.......] 3/10"
 */
export function bar(filled: number, total: number, width = 20): string {
  const safeTotal = Math.max(total, 1);
  const pct = Math.round((filled / safeTotal) * width);
  const count = `${filled}/${total}`;
  if (isTTY()) {
    return `${'█'.repeat(pct)}${'░'.repeat(width - pct)} ${count}`;
  }
  return `[${'#'.repeat(pct)}${'.'.repeat(width - pct)}] ${count}`;
}

/**
 * Horizontal rule with optional centered label.
 * TTY:     "── Label ──────────────────────────"
 * non-TTY: "-- Label ---------------------------"
 */
export function rule(text?: string, width = 55): string {
  const dash = isTTY() ? '─' : '-';
  if (!text) return dash.repeat(width);
  const label = ` ${text} `;
  const sides = Math.max(0, width - label.length);
  const left = Math.floor(sides / 2);
  const right = sides - left;
  return `${dash.repeat(left)}${label}${dash.repeat(right)}`;
}
```

import { describe, it, beforeEach, afterEach } from 'node:test';
import assert from 'node:assert/strict';

// Store original values for cleanup
let originalIsTTY: boolean | undefined;
let originalNoColor: string | undefined;

function setTTY(v: boolean | undefined) {
  (process.stdout as { isTTY: boolean | undefined }).isTTY = v;
}

describe('isColorEnabled', () => {
  beforeEach(() => {
    originalIsTTY = process.stdout.isTTY;
    originalNoColor = process.env.NO_COLOR;
  });
  afterEach(() => {
    setTTY(originalIsTTY);
    if (originalNoColor === undefined) {
      delete process.env.NO_COLOR;
    } else {
      process.env.NO_COLOR = originalNoColor;
    }
  });

  it('returns false when NO_COLOR is set', async () => {
    setTTY(true);
    process.env.NO_COLOR = '1';
    const { isColorEnabled } = await import('../src/cli/ansi.js');
    // NOTE: isColorEnabled reads env at call time — use dynamic import per call or
    // test via c.green() output instead:
    assert.strictEqual(c.green('hello'), 'hello');
  });

  it('c.green wraps with escape code in TTY without NO_COLOR', async () => {
    setTTY(true);
    delete process.env.NO_COLOR;
    const { c } = await import('../src/cli/ansi.js');
    assert.ok(c.green('x').includes('\x1b[32m'));
    assert.ok(c.green('x').includes('\x1b[0m'));
  });

  it('symbol.ok returns "OK" in non-TTY', async () => {
    setTTY(undefined);
    const { symbol } = await import('../src/cli/ansi.js');
    assert.strictEqual(symbol.ok(), 'OK');
  });

  it('symbol.fail returns "FAIL" in non-TTY', async () => {
    setTTY(undefined);
    const { symbol } = await import('../src/cli/ansi.js');
    assert.strictEqual(symbol.fail(), 'FAIL');
  });

  it('bar uses ASCII in non-TTY', async () => {
    setTTY(undefined);
    const { bar } = await import('../src/cli/ansi.js');
    const result = bar(3, 5);
    assert.ok(result.includes('['));
    assert.ok(!result.includes('█'));
  });

  it('box.top uses ASCII in non-TTY', async () => {
    setTTY(undefined);
    const { box } = await import('../src/cli/ansi.js');
    const result = box.top();
    assert.ok(result.startsWith('+'));
    assert.ok(result.includes('-'));
    assert.ok(!result.includes('╔'));
  });

  it('rule returns dashes in non-TTY', async () => {
    setTTY(undefined);
    const { rule } = await import('../src/cli/ansi.js');
    assert.ok(rule().includes('-'));
    assert.ok(!rule().includes('─'));
  });
});
```

Note: Due to ESM module caching, environment variable changes between tests may not reflect in re-imported modules. Consider testing via output inspection (check if the string contains escape codes or not) rather than testing `isColorEnabled()` directly.
